    $(document).ready(function () {
        $('#filesizecheck').on('change', function ()
        {
            myfiles = $(this).val();
            var ext = myfiles.split('.').pop();
            if (ext == "pdf")
            {
                $('#error').css("display", "none");
            }
            else {
                $('#error').html("Only allow .pdf file inputs.");
                $('#error').css("display", "block");
                $('#error').css("color", "red");
                $(this).val(null);
            }
        });
    });
